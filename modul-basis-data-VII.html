Setup Database & Schema

Buat database db_akademik beserta 4 tabel: mahasiswa, mata_kuliah, nilai, dan log_nilai, lalu isi dengan data awal.

1. Buat Database
SQLcreate_database.sql
CREATE DATABASE db_akademik;
USE db_akademik;
2. Buat Tabel
SQLcreate_tables.sql
-- Tabel mahasiswa
CREATE TABLE mahasiswa (
  id_mahasiswa  INT AUTO_INCREMENT PRIMARY KEY,
  nim           VARCHAR(20)  NOT NULL,
  nama          VARCHAR(100) NOT NULL,
  jurusan       VARCHAR(100) NOT NULL,
  angkatan      YEAR         NOT NULL
);

-- Tabel mata_kuliah
CREATE TABLE mata_kuliah (
  id_mk    INT AUTO_INCREMENT PRIMARY KEY,
  kode_mk  VARCHAR(20)  NOT NULL,
  nama_mk  VARCHAR(100) NOT NULL,
  sks      INT          NOT NULL
);

-- Tabel nilai
CREATE TABLE nilai (
  id_nilai      INT AUTO_INCREMENT PRIMARY KEY,
  id_mahasiswa  INT             NOT NULL,
  id_mk         INT             NOT NULL,
  nilai_angka   DECIMAL(5,2)   NOT NULL,
  tanggal_input DATETIME        DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (id_mahasiswa) REFERENCES mahasiswa(id_mahasiswa),
  FOREIGN KEY (id_mk)        REFERENCES mata_kuliah(id_mk)
);

-- Tabel log_nilai (untuk trigger)
CREATE TABLE log_nilai (
  id_log     INT AUTO_INCREMENT PRIMARY KEY,
  id_nilai   INT,
  aksi       VARCHAR(20),
  nilai_lama DECIMAL(5,2),
  nilai_baru DECIMAL(5,2),
  waktu_log  DATETIME DEFAULT CURRENT_TIMESTAMP
);

VIEW

View adalah tabel virtual yang berasal dari query SELECT. Data tidak disimpan secara fisik — view hanya menyimpan definisinya. Ketika dipanggil, hasilnya selalu mengambil data terkini dari tabel aslinya.

AspekPenjelasan
Fungsi utamaMenyimpan query SELECT agar dapat dipanggil seperti tabel biasa.
KelebihanQuery lebih ringkas, tampilan terstruktur, dapat menyembunyikan kolom sensitif.
Kapan digunakanSaat query JOIN atau laporan sering dipakai berulang kali.
CatatanJika data tabel utama berubah, hasil view ikut berubah saat dipanggil.
View Sederhana (WHERE)
VIEWview_mahasiswa_informatika
-- Membuat view mahasiswa jurusan Informatika
CREATE VIEW view_mahasiswa_informatika AS
SELECT nim, nama, jurusan, angkatan
FROM   mahasiswa
WHERE  jurusan = 'Informatika';

-- Memanggil view seperti tabel biasa
SELECT * FROM view_mahasiswa_informatika;
View dengan JOIN (Laporan Nilai)
VIEWview_nilai_mahasiswa
-- View gabungan 3 tabel: nilai + mahasiswa + mata_kuliah
CREATE VIEW view_nilai_mahasiswa AS
SELECT
  m.nim,
  m.nama,
  m.jurusan,
  mk.kode_mk,
  mk.nama_mk,
  mk.sks,
  n.nilai_angka,
  n.tanggal_input
FROM   nilai n
JOIN   mahasiswa  m  ON n.id_mahasiswa = m.id_mahasiswa
JOIN   mata_kuliah mk ON n.id_mk        = mk.id_mk;

-- Tampilkan semua data dari view
SELECT * FROM view_nilai_mahasiswa;

-- Filter dalam view (contoh: hanya BD001)
SELECT * FROM view_nilai_mahasiswa
WHERE  kode_mk = 'BD001'
ORDER BY nilai_angka DESC;
Latihan — View Mahasiswa 2023
VIEWview_mahasiswa_2023
-- Latihan 1: mahasiswa angkatan 2023
CREATE VIEW view_mahasiswa_2023 AS
SELECT nim, nama, jurusan, angkatan
FROM   mahasiswa
WHERE  angkatan = 2023;

-- Latihan 2: laporan nilai (nim, nama, mk, sks, nilai)
CREATE VIEW view_laporan_nilai AS
SELECT
  m.nim,
  m.nama,
  mk.nama_mk,
  mk.sks,
  n.nilai_angka
FROM   nilai n
JOIN   mahasiswa  m  ON n.id_mahasiswa = m.id_mahasiswa
JOIN   mata_kuliah mk ON n.id_mk        = mk.id_mk;
Mengubah dan Menghapus View
VIEWalter & drop
-- Mengubah definisi view (kolom jurusan dihapus)
CREATE OR REPLACE VIEW view_mahasiswa_informatika AS
SELECT nim, nama, jurusan
FROM   mahasiswa
WHERE  jurusan = 'Informatika';

-- Menghapus view
DROP VIEW view_mahasiswa_informatika;
DROP VIEW IF EXISTS view_mahasiswa_informatika; -- aman (tidak error jika belum ada)
💡
View cocok untuk query laporan yang sering dipakai. Namun view bukan pengganti tabel — datanya tetap berasal dari tabel dasar dan tidak menyimpan apapun secara fisik.

TRIGGER

Trigger adalah perintah SQL yang berjalan otomatis ketika terjadi aksi INSERT, UPDATE, atau DELETE pada tabel. Digunakan untuk membuat log, validasi data, atau otomatisasi proses.

Jenis Trigger
BEFORE

BEFORE INSERT

Validasi / ubah nilai sebelum disimpan.

AFTER

AFTER INSERT

Catat log setelah data ditambahkan.

BEFORE

BEFORE UPDATE

Batasi perubahan nilai tertentu.

AFTER

AFTER UPDATE

Catat nilai lama & baru ke log.

BEFORE

BEFORE DELETE

Cegah penghapusan data tertentu.

AFTER

AFTER DELETE

Catat riwayat penghapusan data.

ℹ️
OLD = nilai sebelum perubahan  |  NEW = nilai baru / yang akan disimpan. OLD tersedia di UPDATE & DELETE. NEW tersedia di INSERT & UPDATE.
Trigger AFTER INSERT — Catat Log
TRIGGERtrg_after_insert_nilai
DELIMITER $$

CREATE TRIGGER trg_after_insert_nilai
AFTER INSERT ON nilai
FOR EACH ROW
BEGIN
  -- Catat aksi INSERT ke log_nilai; nilai_lama = NULL karena data baru
  INSERT INTO log_nilai (id_nilai, aksi, nilai_lama, nilai_baru)
  VALUES (NEW.id_nilai, 'INSERT', NULL, NEW.nilai_angka);
END $$

DELIMITER ;

-- Uji coba: tambah nilai baru → cek log_nilai
INSERT INTO nilai (id_mahasiswa, id_mk, nilai_angka)
VALUES (2, 2, 88.00);

SELECT * FROM log_nilai;
Trigger AFTER UPDATE — Catat Perubahan
TRIGGERtrg_after_update_nilai
DELIMITER $$

CREATE TRIGGER trg_after_update_nilai
AFTER UPDATE ON nilai
FOR EACH ROW
BEGIN
  -- OLD = nilai sebelum diubah, NEW = nilai setelah diubah
  INSERT INTO log_nilai (id_nilai, aksi, nilai_lama, nilai_baru)
  VALUES (OLD.id_nilai, 'UPDATE', OLD.nilai_angka, NEW.nilai_angka);
END $$

DELIMITER ;

-- Uji coba: ubah nilai_angka → cek log_nilai
UPDATE nilai
SET    nilai_angka = 92.00
WHERE  id_nilai = 1;

SELECT * FROM log_nilai;
Trigger BEFORE INSERT — Validasi Nilai
TRIGGERtrg_before_insert_nilai
DELIMITER $$

CREATE TRIGGER trg_before_insert_nilai
BEFORE INSERT ON nilai
FOR EACH ROW
BEGIN
  -- Tolak jika nilai di luar rentang 0–100
  IF NEW.nilai_angka < 0 OR NEW.nilai_angka > 100 THEN
    SIGNAL SQLSTATE '45000'
      SET MESSAGE_TEXT = 'Nilai harus berada pada rentang 0 sampai 100';
  END IF;
END $$

DELIMITER ;

-- Uji coba: nilai 120 → akan ditolak trigger
INSERT INTO nilai (id_mahasiswa, id_mk, nilai_angka)
VALUES (1, 3, 120.00); -- Error: Nilai harus berada pada rentang 0 sampai 100
Trigger AFTER DELETE — Catat Data yang Dihapus
TRIGGERtrg_after_delete_nilai
DELIMITER $$

CREATE TRIGGER trg_after_delete_nilai
AFTER DELETE ON nilai
FOR EACH ROW
BEGIN
  -- OLD berisi data yang baru saja dihapus; NEW tidak tersedia di DELETE
  INSERT INTO log_nilai (id_nilai, aksi, nilai_lama, nilai_baru)
  VALUES (OLD.id_nilai, 'DELETE', OLD.nilai_angka, NULL);
END $$

DELIMITER ;

-- Uji coba: hapus nilai → cek log_nilai
DELETE FROM nilai WHERE id_nilai = 5;
SELECT * FROM log_nilai;
⚠️
Terlalu banyak trigger yang saling berkaitan dapat menyulitkan debugging. Pastikan setiap trigger memiliki tanggung jawab yang jelas dan terdokumentasi.
⚙️

STORED PROCEDURE

Stored procedure adalah kumpulan perintah SQL yang disimpan di dalam database. Dipanggil dengan CALL. Cocok untuk proses yang berulang, transaksi, atau laporan.

KomponenPenjelasan
CREATE PROCEDUREMembuat procedure baru.
Parameter INMasukan dari user ke dalam procedure.
Parameter OUTKeluaran dari procedure ke user (variabel).
Parameter INOUTBisa menerima masukan sekaligus mengembalikan keluaran.
CALLMenjalankan stored procedure.
Procedure Tanpa Parameter
PROCtampil_semua_mahasiswa
DELIMITER $$

CREATE PROCEDURE tampil_semua_mahasiswa()
BEGIN
  SELECT * FROM mahasiswa;
END $$

DELIMITER ;

-- Panggil procedure
CALL tampil_semua_mahasiswa();
Procedure dengan Parameter IN
PROCcari_mahasiswa_by_jurusan
DELIMITER $$

CREATE PROCEDURE cari_mahasiswa_by_jurusan(IN p_jurusan VARCHAR(100))
BEGIN
  SELECT nim, nama, jurusan, angkatan
  FROM   mahasiswa
  WHERE  jurusan = p_jurusan;
END $$

DELIMITER ;

-- Panggil dengan argumen
CALL cari_mahasiswa_by_jurusan('Informatika');
Procedure untuk Menambah Data Nilai
PROCtambah_nilai
DELIMITER $$

CREATE PROCEDURE tambah_nilai(
  IN p_id_mahasiswa INT,
  IN p_id_mk        INT,
  IN p_nilai        DECIMAL(5,2)
)
BEGIN
  INSERT INTO nilai (id_mahasiswa, id_mk, nilai_angka)
  VALUES (p_id_mahasiswa, p_id_mk, p_nilai);
END $$

DELIMITER ;

-- Tambah nilai Citra untuk Pemrograman Web = 86
CALL tambah_nilai(3, 2, 86.00);
SELECT * FROM nilai;
Procedure dengan Parameter OUT
PROChitung_jumlah_mahasiswa
DELIMITER $$

CREATE PROCEDURE hitung_jumlah_mahasiswa(OUT total_mahasiswa INT)
BEGIN
  -- Simpan hasil COUNT ke parameter OUT
  SELECT COUNT(*) INTO total_mahasiswa
  FROM   mahasiswa;
END $$

DELIMITER ;

-- Panggil → simpan ke variabel @total
CALL hitung_jumlah_mahasiswa(@total);
SELECT @total AS jumlah_mahasiswa;
Latihan — Tampil Nilai Berdasarkan NIM
PROCtampil_nilai_by_nim
DELIMITER $$

CREATE PROCEDURE tampil_nilai_by_nim(IN p_nim VARCHAR(20))
BEGIN
  SELECT
    m.nim,
    m.nama,
    mk.nama_mk,
    n.nilai_angka,
    konversi_grade(n.nilai_angka) AS grade  -- gunakan function
  FROM  nilai n
  JOIN  mahasiswa   m  ON n.id_mahasiswa = m.id_mahasiswa
  JOIN  mata_kuliah mk ON n.id_mk        = mk.id_mk
  WHERE m.nim = p_nim;
END $$

DELIMITER ;

-- Panggil untuk NIM 23001
CALL tampil_nilai_by_nim('23001');
Latihan — Tambah Data Mahasiswa Baru
PROCtambah_mahasiswa
DELIMITER $$

CREATE PROCEDURE tambah_mahasiswa(
  IN p_nim      VARCHAR(20),
  IN p_nama     VARCHAR(100),
  IN p_jurusan  VARCHAR(100),
  IN p_angkatan YEAR
)
BEGIN
  INSERT INTO mahasiswa (nim, nama, jurusan, angkatan)
  VALUES (p_nim, p_nama, p_jurusan, p_angkatan);
END $$

DELIMITER ;

-- Tambah mahasiswa baru
CALL tambah_mahasiswa('23005', 'Eka Putra', 'Informatika', 2023);
SELECT * FROM mahasiswa;
Latihan — Hitung Jumlah Nilai per Mata Kuliah
PROChitung_nilai_by_mk
DELIMITER $$

CREATE PROCEDURE hitung_nilai_by_mk(
  IN  p_id_mk  INT,
  OUT p_jumlah INT
)
BEGIN
  SELECT COUNT(*) INTO p_jumlah
  FROM   nilai
  WHERE  id_mk = p_id_mk;
END $$

DELIMITER ;

-- Hitung jumlah nilai untuk Basis Data (id_mk = 1)
CALL hitung_nilai_by_mk(1, @jumlah);
SELECT @jumlah AS total_nilai;

FUNCTION

Function adalah objek database yang mengembalikan satu nilai. Dipanggil langsung dalam query SELECT, WHERE, atau ekspresi SQL lainnya. Cocok untuk konversi dan perhitungan.

Perbedaan Stored Procedure Function
Cara memanggil CALL nama_proc() SELECT nama_func()
Nilai balik Opsional (bisa OUT) Wajib satu nilai via RETURN
Kegunaan Proses bisnis, transaksi, query berulang Perhitungan / konversi nilai
Contoh CALL tambah_nilai(...) SELECT konversi_grade(85)
Function Nilai Angka → Grade Huruf
FUNCkonversi_grade
DELIMITER $$

CREATE FUNCTION konversi_grade(p_nilai DECIMAL(5,2))
RETURNS VARCHAR(2)
DETERMINISTIC
BEGIN
  DECLARE grade VARCHAR(2);

  IF      p_nilai >= 85 THEN SET grade = 'A';
  ELSEIF  p_nilai >= 75 THEN SET grade = 'B';
  ELSEIF  p_nilai >= 65 THEN SET grade = 'C';
  ELSEIF  p_nilai >= 50 THEN SET grade = 'D';
  ELSE                        SET grade = 'E';
  END IF;

  RETURN grade;
END $$

DELIMITER ;

-- Test langsung
SELECT konversi_grade(88) AS grade;  -- → A
SELECT konversi_grade(70) AS grade;  -- → B
SELECT konversi_grade(45) AS grade;  -- → E
Function dalam Query SELECT
FUNCpenggunaan di query
-- Tampilkan nilai mahasiswa lengkap beserta grade otomatis
SELECT
  m.nim,
  m.nama,
  mk.nama_mk,
  n.nilai_angka,
  konversi_grade(n.nilai_angka) AS grade
FROM   nilai n
JOIN   mahasiswa   m  ON n.id_mahasiswa = m.id_mahasiswa
JOIN   mata_kuliah mk ON n.id_mk        = mk.id_mk
ORDER BY n.nilai_angka DESC;
Function Status Kelulusan
FUNCstatus_lulus
DELIMITER $$

CREATE FUNCTION status_lulus(p_nilai DECIMAL(5,2))
RETURNS VARCHAR(20)
DETERMINISTIC
BEGIN
  IF p_nilai >= 65 THEN
    RETURN 'Lulus';
  ELSE
    RETURN 'Tidak Lulus';
  END IF;
END $$

DELIMITER ;

-- Gunakan function dari view (view harus sudah dibuat)
SELECT
  nama,
  nama_mk,
  nilai_angka,
  status_lulus(nilai_angka) AS status
FROM view_nilai_mahasiswa;
Latihan — Function Konversi Predikat
FUNCkonversi_predikat
DELIMITER $$

CREATE FUNCTION konversi_predikat(p_nilai DECIMAL(5,2))
RETURNS VARCHAR(30)
DETERMINISTIC
BEGIN
  IF      p_nilai >= 85 THEN RETURN 'Sangat Baik';
  ELSEIF  p_nilai >= 75 THEN RETURN 'Baik';
  ELSEIF  p_nilai >= 65 THEN RETURN 'Cukup';
  ELSE                        RETURN 'Kurang';
  END IF;
END $$

DELIMITER ;

-- Latihan 9: gabungkan function dalam SELECT dari view
SELECT
  nim,
  nama,
  nama_mk,
  nilai_angka,
  konversi_grade(nilai_angka)    AS grade,
  konversi_predikat(nilai_angka) AS predikat,
  status_lulus(nilai_angka)      AS status
FROM view_nilai_mahasiswa
ORDER BY nilai_angka DESC;
💡
Kata kunci DETERMINISTIC menandakan function selalu menghasilkan nilai yang sama untuk input yang sama. Ini penting untuk optimisasi query oleh MySQL.

Melihat & Menghapus Object

Perintah-perintah SQL untuk melihat daftar view, procedure, function, dan trigger yang ada di database, serta menghapus object yang tidak diperlukan.

Melihat Daftar Object
SQLshow_objects.sql
-- Melihat daftar VIEW
SHOW FULL TABLES
WHERE Table_type = 'VIEW';

-- Melihat daftar STORED PROCEDURE
SHOW PROCEDURE STATUS
WHERE Db = 'db_akademik';

-- Melihat daftar FUNCTION
SHOW FUNCTION STATUS
WHERE Db = 'db_akademik';

-- Melihat daftar TRIGGER
SHOW TRIGGERS;

-- Lihat definisi object tertentu
SHOW CREATE VIEW     view_nilai_mahasiswa;
SHOW CREATE PROCEDURE tampil_semua_mahasiswa;
SHOW CREATE FUNCTION konversi_grade;
SHOW CREATE TRIGGER  trg_after_insert_nilai;
Menghapus Object
SQLdrop_objects.sql
-- DROP VIEW
DROP VIEW      view_nilai_mahasiswa;
DROP VIEW IF EXISTS view_mahasiswa_2023;

-- DROP TRIGGER
DROP TRIGGER   trg_after_insert_nilai;
DROP TRIGGER IF EXISTS trg_after_update_nilai;

-- DROP PROCEDURE
DROP PROCEDURE tampil_semua_mahasiswa;
DROP PROCEDURE IF EXISTS tambah_nilai;

-- DROP FUNCTION
DROP FUNCTION  konversi_grade;
DROP FUNCTION IF EXISTS status_lulus;
Ringkasan Perbandingan
Object Definisi Cara Memanggil Contoh Kegunaan
VIEW Tabel virtual dari query SELECT SELECT * FROM nama_view Laporan nilai mahasiswa
TRIGGER SQL otomatis saat INSERT / UPDATE / DELETE Berjalan otomatis Log perubahan nilai
PROCEDURE Kumpulan SQL yang disimpan & dipanggil CALL nama_proc() Tambah nilai, laporan, transaksi
FUNCTION SQL yang menghasilkan satu nilai SELECT nama_func(val) Konversi nilai ke grade
📌
Kapan Menggunakan Apa?
Gunakan View untuk menyederhanakan query laporan yang sering dipakai. Gunakan Trigger untuk otomatisasi seperti pencatatan log atau validasi. Gunakan Stored Procedure untuk menyimpan logika bisnis yang berulang, seperti proses insert transaksi. Gunakan Function untuk konversi atau kalkulasi nilai yang dibutuhkan langsung dalam query SELECT.
